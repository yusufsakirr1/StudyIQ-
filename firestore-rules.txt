// Firebase Console -> Firestore Database -> Rules bölümüne bu kuralları kopyalayın:

// ÖNEMLİ: Firebase Console'a giriş yapın ve aşağıdaki adımları izleyin:
// 1. https://console.firebase.google.com
// 2. Projenizi seçin: studiq-7af4f
// 3. Firestore Database -> Rules
// 4. Aşağıdaki kuralları kopyalayıp yapıştırın
// 5. "Publish" butonuna tıklayın

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users collection - authenticated kullanıcılar kendi verilerini okuyabilir/yazabilir
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Authenticated kullanıcılar diğer verileri okuyabilir
    match /{document=**} {
      allow read: if request.auth != null;
    }
  }
}
